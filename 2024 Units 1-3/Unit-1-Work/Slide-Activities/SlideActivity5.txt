INPUT int duration
INPUT Membership
INPUT dayinput

DEFINE checkweek(day):
    if day in (mon, tue, wed, thurs, fri)
        return 'weekday'
    elif day in (sat, sun)
        return 'weekend'
    else:
        return 'idk'
    
IF membership == 'platinum':
    IF checkweek(dayinput) == 'weekday':
        DISPLAY 'First hour free and after that $2/ hour'
        DISPLAY 'Parking Fees = $' + str((duration-1)*2)
    ELIF checkweek(dayinput) == 'weekend':
        DISPLAY "First 2 hours free and after that $1/ hour"
        DISPLAY 'Parking Fees = $' + str((duration-2)*1)
    ELIF checkweek(dayinput) == 'idk':
        DISPLAY "I'm sorry, but", dayinput ,"is not a valid day"
    ELSE:
        DISPLAY 'Our program is not working right now. Please leave a message. We will ignore it :D '
        INPUT trash

        DISPLAY 'first 2 hours free and after that $2.5/ hour'
        DISPLAY 'Parking Fees = $' + str((float(duration)-2)*2.5)
    ELIF checkweek(dayinput) == 'idk':
        DISPLAY I'm sorry, but", dayinput ,"is not a valid day"
    ELSE:
